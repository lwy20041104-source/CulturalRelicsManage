@RestController
@RequestMapping("/api/ai-chat")
@Api(tags = "AI智能查询")
public class AiChatController {
    
    @Autowired
    private AiChatService aiChatService;
    
    /**
     * AI查询
     */
    @PostMapping("/query")
    @ApiOperation("AI查询")
    @OperationLog(module = "AI查询", type = "查询", content = "AI智能查询")
    public Result<AiRelicQueryResponse> query(@RequestBody AiRelicQueryRequest request) {
        try {
            AiRelicQueryResponse response = aiChatService.query(request);
            return Result.success(response);
        } catch (Exception e) {
            log.error("AI查询失败", e);
            return Result.error("查询失败:" + e.getMessage());
        }
    }
    
    /**
     * 获取会话列表
     */
    @GetMapping("/sessions")
    @ApiOperation("获取会话列表")
    public Result<List<AiChatSessionVO>> getSessions() {
        Long userId = SecurityUtils.getUserId();
        List<AiChatSessionVO> sessions = aiChatService.getUserSessions(userId);
        return Result.success(sessions);
    }
    
    /**
     * 创建新会话
     */
    @PostMapping("/sessions")
    @ApiOperation("创建新会话")
    public Result<Long> createSession(@RequestBody Map<String, String> params) {
        Long userId = SecurityUtils.getUserId();
        String title = params.get("title");
        Long sessionId = aiChatService.createSession(userId, title);
        return Result.success(sessionId);
    }
    
    /**
     * 删除会话
     */
    @DeleteMapping("/sessions/{id}")
    @ApiOperation("删除会话")
    public Result<Void> deleteSession(@PathVariable Long id) {
        aiChatService.deleteSession(id);
        return Result.success();
    }
}

        if (existUser != null) {

        user.setPassword(encryptedPassword);

        user.setStatus(1); // 默认启用

        

        boolean success = sysUserService.save(user);

        if (success) {

            return Result.success(user.getId());

@Aspect
@Component
@Slf4j
public class OperationLogAspect {
    
    @Autowired
    private SysOperationLogService operationLogService;
    
    @Autowired
    private HttpServletRequest request;
    
    /**
     * 环绕通知
     */
    @Around("@annotation(operationLog)")
    public Object around(ProceedingJoinPoint joinPoint, OperationLog operationLog) throws Throwable {
        // 记录开始时间
        long startTime = System.currentTimeMillis();
        
        // 构建日志对象
        SysOperationLog log = new SysOperationLog();
        
        try {
            // 获取当前用户信息
            log.setUserId(SecurityUtils.getUserId());
            log.setUsername(SecurityUtils.getUsername());
            log.setRealName(SecurityUtils.getRealName());
        } catch (Exception e) {
            log.setUsername("匿名用户");
        }
        
        // 设置操作信息
        log.setModule(operationLog.module());
        log.setOperationType(operationLog.type());
        log.setOperationContent(operationLog.content());
        
        // 获取IP地址
        log.setIpAddress(getClientIp(request));
        
        // 获取请求参数
        Object[] args = joinPoint.getArgs();
        if (args != null && args.length > 0) {
            try {
                String params = JSON.toJSONString(args);
                if (params.length() > 2000) {
                    params = params.substring(0, 2000) + "...";
                }
                log.setRequestParams(params);
            } catch (Exception e) {
                log.setRequestParams("参数序列化失败");
            }
        }
        
        Object result = null;
        try {
            // 执行目标方法
            result = joinPoint.proceed();
            
            // 记录成功
            log.setOperationResult("成功");
            
        } catch (Exception e) {
            // 记录失败
            log.setOperationResult("失败");
            log.setErrorMessage(e.getMessage());
            
            throw e;
        } finally {
            // 计算执行时间
            long endTime = System.currentTimeMillis();
            log.setExecutionTime(endTime - startTime);
            
            // 异步保存日志
            operationLogService.saveAsync(log);
        }
        
        return result;
    }
    
    /**
     * 获取客户端IP地址
     */
    private String getClientIp(HttpServletRequest request) {
        String ip = request.getHeader("X-Forwarded-For");
        if (ip == null || ip.length() == 0 || "unknown".equalsIgnoreCase(ip)) {
            ip = request.getHeader("Proxy-Client-IP");
        }
        if (ip == null || ip.length() == 0 || "unknown".equalsIgnoreCase(ip)) {
            ip = request.getHeader("WL-Proxy-Client-IP");
        }
        if (ip == null || ip.length() == 0 || "unknown".equalsIgnoreCase(ip)) {
            ip = request.getRemoteAddr();
        }
        return ip;
    }
}

    }
    
    /**
     * 创建备份
     */
    @PostMapping
    @ApiOperation("创建备份")
    @PreAuthorize("hasRole('ADMIN')")
    @OperationLog(module = "数据备份", type = "新增", content = "创建数据备份")
    public Result<Long> backup(@RequestBody Map<String, Object> params) {
        try {
            Boolean encrypted = (Boolean) params.get("encrypted");
            if (encrypted == null) {
                encrypted = false;
            }
            
            Long backupId = backupService.createBackup(encrypted);
            return Result.success(backupId);
        } catch (Exception e) {
            log.error("备份失败", e);
            return Result.error("备份失败:" + e.getMessage());
        }

    /**
     * 下载备份文件
     */
    @GetMapping("/{id}/download")
    @ApiOperation("下载备份文件")
    @PreAuthorize("hasRole('ADMIN')")
    public void download(@PathVariable Long id, HttpServletResponse response) {
        try {
            SysBackup backup = backupService.getById(id);
            if (backup == null) {
                throw new BusinessException("备份不存在");
            }
            
            File file = new File(backup.getBackupPath());
            if (!file.exists()) {
                throw new BusinessException("备份文件不存在");
            }
            
            response.setContentType("application/octet-stream");
            response.setHeader("Content-Disposition", 
                "attachment; filename=" + URLEncoder.encode(backup.getBackupName(), "UTF-8"));
            
            try (FileInputStream fis = new FileInputStream(file);
                 OutputStream os = response.getOutputStream()) {
                byte[] buffer = new byte[1024];
                int len;
                while ((len = fis.read(buffer)) != -1) {
                    os.write(buffer, 0, len);
                }
                os.flush();
            }
        } catch (Exception e) {
            log.error("下载备份文件失败", e);
        }
    }
    
    /**
     * 恢复备份
     */
    @PostMapping("/{id}/restore")
    @ApiOperation("恢复备份")
    @PreAuthorize("hasRole('ADMIN')")
    @OperationLog(module = "数据备份", type = "修改", content = "恢复数据备份")
    public Result<Void> restore(@PathVariable Long id) {
        try {
            backupService.restoreBackup(id);

        } catch (Exception e) {

    private String backupPath;
    
    @Value("${spring.datasource.url}")
    private String dbUrl;

    private String dbPassword;
    
    @Override
    public Long createBackup(Boolean encrypted) throws Exception {
        // 生成备份文件名
        String timestamp = LocalDateTime.now().format(DateTimeFormatter.ofPattern("yyyyMMdd_HHmmss"));
        String fileName = "backup_" + timestamp + ".sql";
        String filePath = backupPath + File.separator + fileName;
        
        // 确保备份目录存在
        File backupDir = new File(backupPath);
        if (!backupDir.exists()) {
            backupDir.mkdirs();
        }
        
        // 从URL中提取数据库名
        String dbName = extractDbName(dbUrl);
        
        // 执行mysqldump命令
        String command = String.format(
            "mysqldump -u%s -p%s --databases %s --result-file=%s",
            dbUsername, dbPassword, dbName, filePath
        );
        
        Process process = Runtime.getRuntime().exec(command);
        int exitCode = process.waitFor();
        
        if (exitCode != 0) {
            throw new BusinessException("备份失败,退出码:" + exitCode);
        }
        
        // 如果需要加密
        if (encrypted) {
            encryptFile(filePath);
        }
        
        // 获取文件大小
        File file = new File(filePath);
        long fileSize = file.length();
        
        // 创建备份记录
        SysBackup backup = new SysBackup();
        backup.setBackupName(fileName);
        backup.setBackupPath(filePath);
        backup.setBackupSize(fileSize);
        backup.setBackupType(encrypted ? "加密备份" : "普通备份");
        backup.setIsEncrypted(encrypted ? 1 : 0);
        backup.setStatus("成功");
        backup.setCreatorId(SecurityUtils.getUserId());
        backup.setCreatorName(SecurityUtils.getRealName());
        
        this.save(backup);
        
        return backup.getId();
    }

    /**
     * 分页查询图片列表
     */
    @GetMapping("/page")
    public Result<PageResult<ImageLibrary>> getImagePage(
            @RequestParam(defaultValue = "1") Integer pageNum,
            @RequestParam(defaultValue = "20") Integer pageSize,
            @RequestParam(required = false) String keyword) {
        
        Page<ImageLibrary> page = new Page<>(pageNum, pageSize);
        QueryWrapper<ImageLibrary> queryWrapper = new QueryWrapper<>();
        
        if (StringUtils.isNotBlank(keyword)) {
            queryWrapper.like("image_name", keyword)
                       .or()
                       .like("description", keyword);
        }
        
        queryWrapper.orderByDesc("upload_time");
        IPage<ImageLibrary> result = imageLibraryService.page(page, queryWrapper);
        
        return Result.success(new PageResult<>(
            result.getRecords(),
            result.getTotal(),
            result.getPages()
        ));
    }

    /**
     * 上传图片
     */
    @PostMapping("/upload")
    @OperationLog(module = "图片管理", operation = "上传图片")
    public Result<ImageLibrary> uploadImage(@RequestParam("file") MultipartFile file) {
        try {
            // 验证文件类型
            String contentType = file.getContentType();
            if (!contentType.startsWith("image/")) {
                return Result.error("只能上传图片文件");
            }
            
            // 验证文件大小（最大10MB）
            if (file.getSize() > 10 * 1024 * 1024) {
                return Result.error("图片大小不能超过10MB");
            }
            
            // 生成文件名
            String originalFilename = file.getOriginalFilename();
            String extension = originalFilename.substring(originalFilename.lastIndexOf("."));
            String fileName = UUID.randomUUID().toString() + extension;
            
            // 保存文件
            String uploadDir = "uploads/images/";
            File dir = new File(uploadDir);
            if (!dir.exists()) {
                dir.mkdirs();
            }
            
            String filePath = uploadDir + fileName;
            file.transferTo(new File(filePath));
            
            // 保存记录
            ImageLibrary image = new ImageLibrary();
            image.setImageName(originalFilename);
            image.setImagePath(filePath);
            image.setImageUrl("/api/files/" + filePath);
            image.setFileSize(file.getSize());
            image.setUploadTime(LocalDateTime.now());
            image.setUploaderId(SecurityUtils.getUserId());
            image.setUploaderName(SecurityUtils.getUsername());
            
            imageLibraryService.save(image);
            
            return Result.success(image);
        } catch (Exception e) {
            log.error("上传图片失败", e);
            return Result.error("上传失败：" + e.getMessage());
        }
    }

    /**
     * 关联文物
     */
    @PostMapping("/relate")
    @OperationLog(module = "图片管理", operation = "关联文物")
    public Result<Void> relateToRelic(@RequestBody RelicImageRelation relation) {
        relationService.save(relation);
        return Result.success();
    }

    /**
     * 取消关联
     */
    @DeleteMapping("/relate/{relationId}")
    @OperationLog(module = "图片管理", operation = "取消关联")
    public Result<Void> removeRelation(@PathVariable Long relationId) {
        relationService.removeById(relationId);
        return Result.success();
    }

    /**
     * 获取图片的关联文物列表
     */
    @GetMapping("/{imageId}/relics")
    public Result<List<CulturalRelic>> getRelatedRelics(@PathVariable Long imageId) {
        List<CulturalRelic> relics = imageLibraryService.getRelatedRelics(imageId);
        return Result.success(relics);
    }

    /**
     * 删除图片
     */
    @DeleteMapping("/{id}")
    @OperationLog(module = "图片管理", operation = "删除图片")
    public Result<Void> deleteImage(@PathVariable Long id) {
        // 检查是否有关联文物
        long relationCount = relationService.count(
            new QueryWrapper<RelicImageRelation>().eq("image_id", id)
        );
        
        if (relationCount > 0) {
            return Result.error("该图片已关联文物，请先取消关联");
        }
        
        // 删除文件
        ImageLibrary image = imageLibraryService.getById(id);
        if (image != null) {
            File file = new File(image.getImagePath());
            if (file.exists()) {
                file.delete();
            }
        }
        
        // 删除记录
        imageLibraryService.removeById(id);
        return Result.success();
    }

    /**
     * 批量删除图片
     */
    @DeleteMapping("/batch")
    @OperationLog(module = "图片管理", operation = "批量删除图片")
    public Result<Void> batchDelete(@RequestBody List<Long> ids) {
        for (Long id : ids) {
            deleteImage(id);
        }
        return Result.success();
    }
}

    public Result<PageResult<RelicArchive>> getArchivePage(
            @RequestParam(defaultValue = "1") Integer pageNum,
            @RequestParam(defaultValue = "10") Integer pageSize,
            @RequestParam(required = false) String keyword,
            @RequestParam(required = false) String archiveType,
            @RequestParam(required = false) String status) {
        
        Page<RelicArchive> page = new Page<>(pageNum, pageSize);
        QueryWrapper<RelicArchive> queryWrapper = new QueryWrapper<>();
        
        if (StringUtils.isNotBlank(keyword)) {
            queryWrapper.and(wrapper -> wrapper
                .like("archive_code", keyword)
                .or()
                .like("title", keyword));
        }
        
        if (StringUtils.isNotBlank(archiveType)) {
            queryWrapper.eq("archive_type", archiveType);
        }
        
        if (StringUtils.isNotBlank(status)) {
            queryWrapper.eq("status", status);
        }
        
        queryWrapper.orderByDesc("create_time");
        IPage<RelicArchive> result = archiveService.page(page, queryWrapper);
        
        return Result.success(new PageResult<>(
            result.getRecords(),
            result.getTotal(),
            result.getPages()
        ));
    }

    @PostMapping("/{id}/documents")
    @OperationLog(module = "档案管理", operation = "上传文档")
    public Result<ArchiveDocument> uploadDocument(
            @PathVariable Long id,
            @RequestParam("file") MultipartFile file,
            @RequestParam(required = false) String documentType) {
        
        try {
            // 生成文件名
            String originalFilename = file.getOriginalFilename();
            String extension = originalFilename.substring(originalFilename.lastIndexOf("."));
            String fileName = UUID.randomUUID().toString() + extension;
            
            // 保存文件
            String uploadDir = "uploads/archives/";
            File dir = new File(uploadDir);
            if (!dir.exists()) {
                dir.mkdirs();
            }
            
            String filePath = uploadDir + fileName;
            file.transferTo(new File(filePath));
            
            //             // 保存记录
            ArchiveDocument document = new ArchiveDocument();
            document.setArchiveId(id);
            document.setDocumentType(documentType);
            document.setFileName(originalFilename);
            document.setFilePath(filePath);
            document.setFileSize(file.getSize());
            document.setUploadTime(LocalDateTime.now());
            document.setUploaderId(SecurityUtils.getUserId());
            document.setUploaderName(SecurityUtils.getUsername());
            
            documentService.save(document);
            
            // 记录历史
            ArchiveHistory history = new ArchiveHistory();
            history.setArchiveId(id);
            history.setOperationType("UPLOAD_DOCUMENT");
            history.setOperationDesc("上传文档：" + originalFilename);
            history.setOperatorId(SecurityUtils.getUserId());
            history.setOperatorName(SecurityUtils.getUsername());
            historyService.save(history);
            
            return Result.success(document);
        } catch (Exception e) {
            log.error("上传文档失败", e);
            return Result.error("上传失败：" + e.getMessage());
        }
    }

const handleCommand = async (command, row) => {
  switch (command) {
    case 'publish':
      await publishArchive(row.id)
      break
    case 'archive':
      await archiveDocument(row.id)
      break
    case 'exportPdf':
      exportArchive(row.id, 'pdf')
      break
    case 'exportWord':
      exportArchive(row.id, 'word')
      break
    case 'delete':
      await deleteArchive(row.id)
      break
  }
}

const formatFileSize = (bytes) => {
  if (bytes === 0) return '0 B'
  const k = 1024
  const sizes = ['B', 'KB', 'MB', 'GB']
  const i = Math.floor(Math.log(bytes) / Math.log(k))
  return Math.round(bytes / Math.pow(k, i) * 100) / 100 + ' ' + sizes[i]
}
